isArray.js

import { _toString } from './util';

/**
 * @module
 * @description 判断一个变量是否是数组
 * @function isArray
 * @param {*} arr 要判断的变量值
 * @returns {boolean} 返回是否是数组的结果
 * @since 0.0.1
 * @example
 * isArray({})
 * // return false
 *
 * isArray([])
 * // return true
 *
 * isArray(new Array())
 * // return true
 *
 * @exports isArray
 * @requires util.js
 */
export default function isArray(arr) {
  return _toString.call(arr) === '[object Array]';
}
