/**
 * @module
 * @description 判断变量是否已经被定义
 * @function isDef
 * @param {*} arr 要判断的变量值
 * @returns {boolean} 返回判断的结果，如果变量已定义，返回true，反之返回false
 * @since 0.0.1
 * @example
 * isDef('')
 * // return true
 *
 * isDef(undefined)
 * // return false
 *
 * isDef(null)
 * // return false
 *
 * @exports isDef
 */
export default function isDef(val) {
  return val !== undefined && val !== null;
}
