import isArray from './isArray';
import isUndef from './isUndef';

/**
 * @module
 * @description 判断两个数组是否存在相同的对象 根据对象中的某一字段进行判断
 * @function isExistSameObject
 * @param {Object[]} arr1 第一个数组
 * @param {Object[]} arr2 第二个数组
 * @param {string} key 进行判断的依据字段
 * @returns {boolean | undefined} undefined 表示参数不符合规则 true 表示存在相同的对象 false表示不存在相同对象
 * @since 0.0.1
 * @example
 * isExistSameObject([{id: 1, value: 1},{id: 2, value: 2}], [{id: 2, value: 2}, {id: 3, value: 3}], 'id')
 * // return true
 *
 * isExistSameObject([{id: 1, value: 1}], [{id: 2, value: 2}], 'id')
 * // return false
 *
 * isExistSameObject([{id: 1, value: 1 }], {}, 'id')
 * // return undefined
 *
 * @exports isExistSameObject
 * @requires isArray
 * @requires isUndef
 */
export default function isExistSameObject(arr1, arr2, field) {
  if (!isArray(arr1) || !isArray(arr2) || isUndef(field)) return undefined;
  let flag = false;
  for (let i = 0; i < arr1.length; i++) {
    let index = arr2.findIndex((item) => item[field] === arr1[i][field]);
    if (index >= 0) {
      flag = true;
      break;
    }
  }
  return flag;
}
