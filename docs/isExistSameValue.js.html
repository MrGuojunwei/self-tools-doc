isExistSameValue.js

import isArray from './isArray';

/**
 * @module
 * @description 判断两个数组中是否存在相同的值
 * @function isExistSameValue
 * @param {Array} arr1 第一个数组 数组元素为基本类型值
 * @param {Array} arr2 第二个数组 数组元素为基本类型值
 * @returns {boolean|undefined} 返回判定结果 undefined表示参数不符合规范 true表示存在相同的值 false表示不存在相同值
 * @since 0.0.1
 * @example
 * isExistSameValue([1,2,3], ['1',2,'3'])
 * // return true
 *
 * isExistSameValue(['1','2','3'], [1,2,3])
 * // return false
 *
 * isExistSameValue([1,2,3], '123')
 * // return undefined
 *
 * @exports isExistSameValue
 * @requires isArray
 */
export default function isExistSameValue(arr1, arr2) {
  if (!isArray(arr1) || !isArray(arr2)) return undefined;
  let flag = false;
  for (let i = 0; i < arr1.length; i++) {
    if (arr2.includes(arr1[i])) {
      flag = true;
      break;
    }
  }

  return flag;
}
