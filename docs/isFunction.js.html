isFunction.js

/**
 * @module
 * @description 判断变量是否是函数
 * @function isFunction
 * @param {*} fn 需要进行判断的变量
 * @returns {boolean} 变量是函数类型 返回值为true，否则返回false  
 * @since 0.0.1
 * @example
 * const a = new Function();
 * const b = function () {};
 * const c = new Object();
 * 
 * isFunction(a);
 * // -> true
 * 
 * isFunction(b);
 * // -> true
 * 
 * isFunction(c);
 * // -> false
 * 
 * @exports isFunction
 */
export default function isFunction(fn) {
  return typeof fn === 'function';
}
