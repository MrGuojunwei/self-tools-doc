isNotSpaceString.js

import isString from './isString';

/**
 * @module
 * @description 判断一个值是否是非空字符串, 注意：仅包含空格的字符将被认为是空字符串
 * @function isNotSpaceString
 * @param {String} str 需要判断的值
 * @returns {boolean} 需判断的值是字符串且非空时返回true，否则返回false
 * @since 0.0.1
 * @example
 *
 * isNotSpaceString('')
 * // -> false
 *
 * isNotSpaceString('   ')
 * // -> false
 *
 * isNotSpaceString('str')
 * // -> true
 *
 * isNotSpaceString(new String(123));
 * // -> true
 *
 * @exports isNotSpaceString
 * @requires isString
 */
export default function isNotSpaceString(str) {
  return isString(str) && str.trim() !== '';
}
