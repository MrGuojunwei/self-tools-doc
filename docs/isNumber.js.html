isNumber.js

/**
 * @module
 * @description 判断一个值是否是数值类型
 * @function isNumber
 * @param {any} num 需要进行判断的值
 * @returns {boolean} true 表示该值是数值类型 false表示该值不是数值类型
 * @since 0.0.1
 * @example
 * isNumber(0)
 * // return true
 *
 * isNumber('0');
 * // return false
 *
 * isNumber(Number('12'))
 * // return true
 * 
 * isNumber(new Number(3))
 * // return false
 *
 * @exports isNumber
 */
function isNumber(num) {
    return typeof num === 'number';
}

export default isNumber;
