/*
 * @Description: isString 判断变量是否是字符串
 * @Author: 郭军伟
 * @Date: 2020-10-13 16:39:15
 * @LastEditors: 郭军伟
 * @LastEditTime: 2020-11-04 10:39:31
 */
import { _toString } from './util';

/**
 * @module
 * @description 判断一个变量是否是字符串类型
 * @function isString
 * @param {any} str 需要判断的变量   
 * @returns {boolean} 字符串类型返回true 否则返回false 
 * @since 0.0.1
 * @example
 * isString('')
 * // return true
 * 
 * isString(new String())
 * // return true
 * 
 * isString({})
 * // return false
 * 
 * isString(123)
 * // return false
 * 
 * @exports isString
 * @requires _toString
 */
export default function isString(str) {
  return _toString.call(str) === '[object String]';
}
