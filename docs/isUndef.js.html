/*
 * @Description: isUndef 判断变量是否是未定义
 * @Author: 郭军伟
 * @Date: 2020-10-13 16:38:37
 * @LastEditors: 郭军伟
 * @LastEditTime: 2020-11-04 14:03:38
 */
/**
 * @module
 * @description 判断一个变量是否是无值状态，undefined或null认为是无值状态
 * @function isUndef
 * @param {any} val 需要进行判断的变量
 * @returns {boolean} 无值返回true  否则返回false
 * @since 0.0.1
 * @example
 * isUndef(null)
 * // return true
 * 
 * isUndef(undefined)
 * // return true
 * 
 * isUndef([])
 * // return false
 * 
 * isUndef('')
 * // return false
 * 
 * 
 * @exports isUndef
 */
export default function isUndef(val) {
  return val === undefined || val === null;
}
