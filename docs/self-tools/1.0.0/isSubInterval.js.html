import isArray from './isArray';

/**
 * @module
 * @description 判断一个包含两个数值类型的数组区间是否在另一个包含两个数值类型的数组区间中
 * @function isSubInterval
 * @param {Array} parentInterval 父区间数组
 * @param {Array} subInterval 子区间数组
 * @return {boolean | undefined} undefined参数不符合要求  true表示子数组区间在父数组区间中 false表示子数组区间不在父数组区间中
 * @since 0.0.1
 * @example
 * let a = [3,7];
 * let b = [4,6];
 * let c = [1,6];
 * let d = [6,2];
 * let e = [1,9];
 * 
 * isSubInterval(a, []);
 * // return undefined
 *
 * isSubInterval(a, b);
 * // return true
 *
 * isSubInterval(a, c);
 * // return false
 *
 * isSubInterval(a, d);
 * // return false
 *
 * isSubInterval(e, d);
 * // return true
 *
 * @exports isSubInterval
 * @requires isArray
 */
export default function isSubInterval(parentInterval, subInterval) {
  if (
    !isArray(parentInterval) ||
    !isArray(subInterval) ||
    parentInterval.length !== 2 ||
    subInterval.length !== 2
  ) {
    return undefined;
  }
  // 将数组进行从小到大排序
  parentInterval = parentInterval.sort((a, b) => a - b);
  subInterval = subInterval.sort((a, b) => a - b);

  return (
    subInterval[0] >= parentInterval[0] && subInterval[1] <= parentInterval[1]
  );
}
